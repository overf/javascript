Selector
Javascript에서 DOM을 통해 HTML 문서의 속성들을 불러오는것을 조금 더 쉽게 제어할 수 있고, 그 범위가 넓어서 다양한 선택자가 존재한다.

javascript selector

구분 getElementsBy* querySelector*
id document.getElementById(‘foo’) (모든브라우저지원) document.querySelector(‘#foo’)
tag document.getElementsByTagName(‘li’) (모든브라우저지원) document.querySelectorAll(‘li’)
class document.getElementsByClassName(‘bar’) (ie9 이상지원) document.querySelector(‘.bar’)
속성(name=foo) document.getElementsByName(‘foo’) (모든브라우저지원) document.querySelectorAll(‘[name=foo]’)
querySelector = 하나의요소 선택시 사용(ie8 이상지원) , querySelectorAll = 여러개의요소 선택시 사용(ie9 이상지원) 
                            
//요소의 id 속성값으로 선택
var elem = document.getElementById('아이디명');


//요소의 class 속성값으로 선택
var elem = document.getElementsByClassName('클래스명');

//요소의 tagName 속성값으로 선택
var elem = document.getElementsByTagName('태그명');

//요소의 name 속성값으로 선택
var elem = document.getElementsByName('이름명');


//querySelector
document.querySelector('#foo'); //id 선택
document.querySelector('.foo'); //class 선택
document.querySelector('li'); //tag 선택
document.querySelector('[name=foo]'); //name 선택

//querySelectorAll
document.querySelectorAll('.foo'); //class 선택
document.querySelectorAll('li'); //tag 선택
document.querySelectorAll('[name=foo]'); //name 선택

//복수요소 선택시 활용방법
var elems = document.querySelectorAll('클래스명');
//or
var elem = document.getElementsByClassName('클래스명');

for (var i=0; i < elems.length; i++){
    elems[i].style.backgroundColor = 'blue';
}
//elems[인덱스값] 선택된 요소를 인덱스값으로 검색하여 사용가능

DOM event

브라우저는 이벤트를 감지할 수 있으며 이벤트 발생 시에는 통지해 준다. 이 과정을 통해 사용자와 웹페이지는 상호작용(Interaction)이 가능하게 된다.

style 참고 링크 
                            
//객체 선택후 style 적용

//배경색 변경
var elem = document.getElementById('colorDiv');
elem.style.backgroundColor = "red";


//left 값 변경
var elem = document.getElementById('colorDiv');
elem.style.left = "100px";
