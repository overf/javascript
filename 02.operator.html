Operator
연산자(Operators)는 하나 혹은 그 이상의 값을 하나의 값으로 만들 때 사용한다.

1. 산술 연산자 (Arithmetic Operators)

Operator Description
+ 덧셈
- 뺄셈
* 곱셈
/ 나눗셈
% 나머지
++ 증가
-- 감소 
                    
var value_1 = 5;
var value_2 = 2;
var resualt;

resualt = value_1 + value_2;  // 7
resualt = value_1 - value_2;  // 3
resualt = value_1 * value_2;  // 10
resualt = value_1 / value_2;  // 2.5
resualt = value_1 % value_2;  // 1
resualt = value_1++;    // 5 선대입후증가
resualt = ++value_1;    // 7 선증가후대입
resualt = value_1--;    // 7 선대입후감소
resualt = --value_1;    // 5 선감소후대입

/*
+ 연산자는 덧셈 연산과 문자열 연결 연산을 수행한다.
연산 대상이 모두 숫자인 경우 : 덧셈 연산
그 외의 경우 : 문자열 연결 연산
*/
var str1 = "5" + 5;      // '55'
var str2 = 5 + "5";      // '55'
var str3 = "Hello" + 5;  // 'Hello5'

3. 비교 연산자 (Comparison Operators)

Operator Description
== 동등비교 (loose equality)
=== 일치비교 (strict equality)
!= 부등비교
!== 불일치비교
> , < , >= , <= 관계비교
? 삼항 연산자 
                    
var x = 5

x == 5    //x가 5일경우 true
x == '5'  //x가 5일경우 true
x == 8    //x가 8일경우 false

x === 5   //x가 5이고 데이터타입이 일치할경우 true
x === '5' //x가 5이고 데이터타입이 일치할경우 false

x != 8    //x가 8이 아닐경우 true
x != 5    //x가 5가 아닐경우 false
x != '5'  //x가 5이 아닐경우 false

x !== 8   // true
x !== 5   // false
x !== '5' // true

x > 0     // true
x > 5     // false
x > 8     // false

x < 0     // false
x < 5     // false
x < 8     // true

x >= 0    // true x가 0보다 크거나 같을때
x >= 5    // true x가 5보다 크거나 같을때
x >= 8    // false x가 8보다 크거나 같을때

x <= 0    // false x가 0보다 작거나 같을때
x <= 5    // true x가 5보다 작거나 같을때
x <= 8    // true x가 8보다 작거나 같을때

//삼항연산자(ternary operator)
/*
 ?: 연산자는 if...else 문의 단축형으로 사용할 수 있으며 일반적으로 if...else 문을 사용하면 너무 복잡해지는 식에서 사용합니다.
*/
test ? expression1 : expression2
조건    참일경우실행    거짓일경우실행


var now = new Date();
var greeting = "Good" + ((now.getHours() > 17) ? " evening." : " day.");
/* getHours() > 17  시간이 17시보다 클경우 참이라면 evening 거짓이라면 day.*/

4. 논리 연산자 (Logical Operators)

Operator Description
&& and
|| or
! not 
                    
// && (논리곱) 연산자
var a1 =  true && true;     // t && t returns true
var a2 =  true && false;    // t && f returns false
var a3 = false && true;     // f && t returns false
var a4 = false && (3 == 4); // f && f returns false
var a5 = "Cat" && "Dog";    // t && t returns Dog
var a6 = false && "Cat";    // f && t returns false
var a7 = "Cat" && false;    // t && f returns false

// || (논리 합) 연산자
var o1 =  true || true;     // t || t returns true
var o2 = false || true;     // f || t returns true
var o3 =  true || false;    // t || f returns true
var o4 = false || (3 == 4); // f || f returns false
var o5 = "Cat" || "Dog";    // t || t returns Cat
var o6 = false || "Cat";    // f || t returns Cat
var o7 = "Cat" || false;    // t || f returns Cat

// ! (논리 부정) 연산자
var n1 = !true;  // false
var n2 = !false; // true
var n3 = !"Cat"; // false

/*예)*/
var ex = 5;
if(ex > 1 && ex < 5){

}

if(ex > 1 || ex < 5){
    /*ex가 1보다 크고 ||(또는) ex가 5보다 작을때 실행*/
}

5. 타입 연산자(Type Operators)

Operator Description
typeof 변수의 자료형을 문자열로 반환한다. null과 배열의 경우 object, 함수의 경우 function를 반환하는 것에 유의하여야 한다.
instanceof 객체가 동일 객체형의 인스턴스이면 true를 반환한다. 
                    
typeof "John"                 // returns string
typeof 3.14                   // returns number
typeof NaN                    // returns number
typeof false                  // returns boolean
typeof [1, 2, 3, 4]           // returns object
typeof {name:'John', age:34}  // returns object
typeof new Date()             // returns object
typeof function () {}         // returns function
typeof myCar                  // returns undefined (if myCar is not declared)
typeof null                   // returns object

//예)
var num = 7;
console.log(typeof num); // number


function Person(){}
var me = new Person()
me instanceof Person // true
