VAR & FUNCTION

var
function

ex

                    
        function foo() {
          var x = 1, y = 2;
          console.log(x + y);
        }
        foo();

        // 객체리터럴
        var obj = {
          x: 1,
          y: 2
        };
        console.log(obj.x + obj.y);

        // 흐름 제어 구문(control flow statement)
        var x = 0;
        while (x < 10) {
          x++;
        }
        console.log(x);
